1. Create a DB
sudo -u postgres createdb _DBNAME_

2. Delete a DB
sudo -u postgres dropdb _DBNAME_

3. Data Base Backup (comple)
sudo -u postgres pg_dump _DBNAME_ > _DBNAME_.sql

4. Backing singe table 
sudo -u postgres pg_dump -t _TABLENAME_ _DBNAME_ > _DBNAME_TABLENAME_.sql

5. Show info avbout all cluster
pg_lsclusters

6. Log in to a DB
sudo -u postgres psql _DB_NAME_

CREATE DATABASE myproject; # Create DB for project
CREATE USER myprojectuser WITH PASSWORD 'password'; # Create USer for DB

# Configure User for Unicode 
ALTER ROLE myprojectuser SET client_encoding TO 'utf8';
ALTER ROLE myprojectuser SET default_transaction_isolation TO 'read committed';
ALTER ROLE myprojectuser SET timezone TO 'UTC';

# access rights
GRANT ALL PRIVILEGES ON DATABASE myproject TO myprojectuser;
